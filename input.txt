<!DOCTYPE html>
<html>
  <head>
    <title>Prompt Example</title>
    <script>
      function askUser() {
        const userInput = prompt("Please enter your name:");
        if (userInput !== null) {
          document.getElementById("output").textContent = "Hello, " + userInput + "!";
        }
      }
    </script>
  </head>
  <body>
    <button onclick="askUser()">Click to Enter Name</button>
    <p id="output"></p>
  </body>
</html>
