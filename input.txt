from playwright.sync_playwright import sync_playwright

with sync_playwright() as p:
    browser = p.chromium.launch()
    page = browser.new_page()

    page.on("dialog", lambda dialog: dialog.accept(prompt_text="Your Input Value"))
    page.goto("your_website_with_prompt")
    page.locator("button_triggering_prompt").click()

    browser.close()
