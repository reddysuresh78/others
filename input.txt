

from pypdf import PdfReader, PdfWriter

# Open an existing PDF
pdf_path = "/Users/sureshreddy/NotBackedupDocs/Downloads_Nobackup/sarada_hyd_delhi.pdf"
reader = PdfReader(pdf_path)
writer = PdfWriter()

# Copy pages
for page in reader.pages:
    writer.add_page(page)

# Add custom metadata
writer.add_metadata({
    "/Title": "Custom Metadata Example",
    "/Author": "John Doe",
    "/Subject": "PDF Metadata Editing",
    "/Keywords": "Python, PDF, Metadata",
    "/CustomProperty1": "My Custom Value",
    "/CustomProperty2": "Another Custom Entry updated"
})

# Save the updated PDF
with open("updated_example.pdf", "wb") as output_pdf:
    writer.write(output_pdf)

print("Custom metadata added successfully!")
